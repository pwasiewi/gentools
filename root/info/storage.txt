# Disk, Filesystem & Storage — SSD alignment, ext4/btrfs, LVM, swap

## SSD Alignment — Samsung EVO (1536 kB EBS, 8 kB page size)
# stride = page/block = 8kB/4kB = 2
# stripe-width = EBS/block = 1536kB/4kB = 384
mkfs.ext4 -b 4096 -E stride=2,stripe-width=384 /dev/sdXX

# Partitioning with gdisk aligned to 3072 sectors (= 1536 kB)
# gdisk /dev/sda → x → L → 3072 → m → n ...

# Generic 512 MB erase block SSD
mkfs.ext4 -E stripe_width=128,stride=128 /dev/sdxx
tune2fs -O has_journal -o journal_data_writeback /dev/sdxx

# References:
# https://busybox.net/~aldot/mkfs_stride.html
# https://wiki.gentoo.org/wiki/SSD

## ext3
mkfs.ext3 /dev/sdb1
tune2fs -O has_journal -o journal_data_writeback /dev/sdb1
mount -t ext3 /dev/sdb1 /mnt/dc -o noatime,data=writeback
/sbin/mkfs.ext3 -j -m 1 -O dir_index,filetype,sparse_super /dev/mapper/encrypted

## ext4
mount -t ext4 /dev/sdxx /mnt/dc -o noatime,extents
tune2fs -Ohas_journal,extent,dir_index,filetype /dev/sdxx
# Reiser4 (legacy)
mkfs.reiser4 -o create=ccreg40 /dev/xxx
mkfs.reiser4 -o create=ccreg40,compress=gzip1 /dev/xxx

## btrfs
mount -o ssd,noatime,nodiratime,compress=zstd /dev/sdxx /mnt
btrfs filesystem defragment -r -v -czstd /
btrfs filesystem resize 99G /mnt/key
btrfs filesystem resize max /mnt/key

## LVM
# https://www.redhat.com/sysadmin/resize-lvm-simple
# https://dannyda.com/2020/05/10/how-to-delete-remove-local-lvm-from-proxmox-ve-pve-and-some-lvm-basics-commands/
pvcreate --dataalignment 512k /dev/sdxx
pvcreate --dataalignment 1536k /dev/sdxx   # Samsung EVO alignment
pvcreate /dev/sdb
pvdisplay; pvscan; pvs

vgcreate my_volume_group /dev/sdb /dev/sdc
vgdisplay; vgs

lvcreate -L 5G -n 5gig my_volume_group
lvdisplay; lvs
mkfs.ext4 /dev/my_volume_group/5gig
mount -t ext4 /dev/my_volume_group/5gig /home
lvremove /dev/my_volume_group/5gig

# Extend existing volume
sgdisk -N 1 /dev/sdc
pvcreate /dev/sdc1
vgextend pve /dev/sdc1
lvextend -l +100%FREE /dev/pve/data
lvresize -L100G /dev/lvm/root
lsblk

# Proxmox thin pool rebuild
# https://forum.proxmox.com/threads/resizing-pve-data.30506/
lvremove /dev/pve/data -y
lvcreate -L 200G -n data pve
lvconvert --type thin-pool pve/data

# /etc/lvm/lvm.conf recommended options:
# md_chunk_alignment = 1
# data_alignment_detection = 1
# data_alignment = 0
# data_alignment_offset_detection = 1

## Swap file
dd if=/dev/zero of=/swapfile bs=1M count=100 && chmod 600 /swapfile && mkswap /swapfile
# /etc/fstab:
# /swapfile   none   swap   defaults   0   0
swapon -a

## SMART monitoring
# https://www.thomas-krenn.com/en/wiki/Analyzing_a_Faulty_Hard_Disk_using_Smartctl
e smartmontools sdparm
for j in Raw_Read_Error_Rate Reallocated_Sector_Ct Seek_Error_Rate Runtime_Bad_Block End-to-End_Error Spin_Retry_Count Wear_Range_Delta Command_Timeout Current_Pending_Sector Offline_Uncorrectable Power_On_Hours; do
  for i in a b c d; do echo /dev/sd$i; smartctl -a /dev/sd$i | grep $j; done
done
smartctl -a /dev/sdb -d cciss,2

## USB ext4 — non-journaled (good for flash drives)
dd bs=440 count=1 conv=notrunc if=/usr/share/syslinux/mbr.bin of=/dev/sde
mkfs.ext4 /dev/sde1 -L Gen8
tune2fs /dev/sde1 -i0
tune2fs /dev/sde1 -m0 -O sparse_super
tune2fs -O ^has_journal /dev/sde1

## I/O tuning
ionice -c2 -n7 nice -n 19 cp
